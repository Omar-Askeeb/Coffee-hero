4) In _ContinueBar onTap, replace placeholder with:

   onTap: () async {
     final ok = await confirmPlaceOrder(context, address: OrdersStore.defaultAddress);
     if (ok != true) return;

     final cart = CartStore.instance;
     if (cart.lines.isEmpty) return;

     final order = await OrdersStore.instance.createFromCart(
       cartLines: cart.lines,
       address: OrdersStore.defaultAddress,
     );
     await cart.clear();

     if (!context.mounted) return;
     Navigator.of(context).pushReplacementNamed(AppRoutes.trackOrder, arguments: order.id);
   },
